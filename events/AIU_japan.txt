				type = protect
				id = "ANH"
				value = 100
				}
				add_ai_strategy = {
				type = alliance
				id = "JAP"
				value = 100
				}
			}
			ANH = {
				add_ai_strategy = {
				type = alliance
				id = "CHF"
				value = 200
				}
				add_ai_strategy = {
				type = alliance
				id = "JAP"
				value = 100
				}
			}
			ANH = { country_event = { days = 2 id = aiu_japan.5 } }
			CHF = { country_event = { days = 2 id = aiu_japan.5 } }
		}
	}

		name = aiu_japan.19.b

			OR = {
				has_country_flag = ANH_agreed

				has_country_flag = ANH_agreed
				has_country_flag = CHF_agreed
				}
			}
		}
		ai_chance = {
			base = 10
		}
		add_political_power = -25
	}
	
	
	option = { # neither agreed
		name = aiu_japan.19.c
		trigger = {
			NOT = { 
				OR = {
				has_country_flag = ANH_agreed
				has_country_flag = CHF_agreed
				}
			}
		}
		ai_chance = {
			base = 10
		}
		add_political_power = -50
	}
}
