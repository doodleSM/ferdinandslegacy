country_event = {
	id = austria.1 #Oil Imports from Romania
	title = austria.1.t
	desc = austria.1.d
	picture = GFX_austrian_army

	fire_only_once = yes

	triggers = {
		has_completed_focus = GOR_romanian_oil_imports

		AND = {
			country_exists = GOR
			country_exists = ROM
		}

		NOT = {
			has_war_with = ROM
		}
	}
	option = {
		name = austria.1.a
		add_ideas = romanian_oil_imports
		ROM = {
			add_ideas = romanian_oil_exports
		}

		create_import = {
			resource = oil
			amount = 15
			exporter = ROM
		}
	}
	option = {
		name austria.1.b
		ROM = {
			country_event = {
				id = romania.1
				days = 5
			}
		}
	}
}
